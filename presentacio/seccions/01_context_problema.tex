% ---  DIAPOSITIVA 3 — El Problema: Fi d'una Era  ---
\begin{frame}{Problema}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \begin{alertblock}{ActiveX (anys 90)}
        \begin{itemize}
          \item Dependent de \textbf{IE11} (tancat juny 2022).
          \item Incompatible: macOS, Linux, Chrome.
          \item Vulnerabilitats \textbf{OWASP}.
        \end{itemize}
      \end{alertblock}

      \vspace{0.5em}
      \begin{itemize}
        \item Arquitectura molt \textbf{rígida}.
        \item Manteniment i actualitzacions manuals/difícils. 
        \item UX obsoleta.
        \item Fre al \textit{time-to-market}.
      \end{itemize}
    \end{column}

    \begin{column}{0.40\textwidth}
      \begin{block}{Mercat}
        \small Accés des de qualsevol navegador i dispositiu. Exigeix web modern.
      \end{block}

      \vspace{0.8em}
      \begin{block}{Competència}
        \small Interfícies web natives guanyaven terreny.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

  \note{

    El tercer punt és la innovació diferencial: que l'aplicació sigui escalable. Si demà surten nous models de firmware amb noves opcions, el frontend s'adapti sol, sense que cap enginyer de software hagi de tocar el codi de la web.
