      \vspace{0.3em}

export interface StreamingStrategy {
  connectionState: Signal<
    'connecting'|'connected'|
    'error'|'idle'>;
  errorMessage: Signal<string|null>;
  startStream(
    video: HTMLVideoElement,
    cam: string): Promise<void>;
  closeStream(): void;
}
// Angular injecta WebRTCService o HlsService
// en funcio del protocol actiu \end{lstlisting}
    \end{column}

    \begin{column}{0.46\textwidth}

      \begin{itemize}
        \item El component crida \texttt{startStream()} \textbf{sempre igual}.
        \item Angular injecta \texttt{WebRTCService} o \texttt{HlsService} \textbf{sense que el component ho sàpiga}.
        \item Nou protocol futur $\Rightarrow$ \textbf{zero canvis} al component.
      \end{itemize}

% ---  DIAPOSITIVA — Gravacions 2/3  ---
\begin{frame}{Gravacions \small\textit{2/3}}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Playlist HLS:}
      \begin{itemize}
        \item Angular demana \texttt{\ldots?token=\ldots\&start=\ldots\&end=\ldots}
        \item \texttt{la-video-engine} busca blocs \textbf{FS2} del rang i composa la playlist
        \item Tall de gravació ($>$3 s) $\to$ \texttt{\#EXT-X-DISCONTINUITY} $\to$ descodificador no es congela.
      \end{itemize}

      \vspace{0.8em}
    \end{column}

    \begin{column}{0.48\textwidth}
      \textbf{Per cada segment demanat:}
      \begin{enumerate}
        \item Llegir bloc \textbf{FS2} del disc (buffer 8 MB).
        \item Blocs \textbf{FS2} $\to$ \textbf{MPEG-TS}: bitstream intacte, CPU $\approx$ 0.
        \item \textbf{H.265}: si falten SPS/PPS/VPS al buffer, el \texttt{la-video-engine} les recupera del keyframe anterior i les injecta.
        \item Servir \texttt{segment.ts} al navegador.
      \end{enumerate}
    \end{column}
  \end{columns}
\end{frame}

% ---  DIAPOSITIVA — Alarmes 2/3  ---
\begin{frame}{Alarmes \small\textit{2/3}}
  \begin{columns}[T]
    \begin{column}{0.46\textwidth}
      \small
      \textbf{\texttt{la-core-alarm} — pipeline de 3 threads:}
      \begin{enumerate}
        \item \textbf{D-Bus}: escolta \texttt{ALARM\_NOTIFY} $\to$ cua \textit{thread-safe}.
        \item \textbf{Persistència}: escriu via \textbf{gRPC} a \texttt{la-db-engine}.
        \item \textbf{Emissió}: difon JSON a tots els clients via \textbf{WSS}.
      \end{enumerate}
      \vspace{0.3em}
      \begin{itemize}
        \item Fils \textbf{desacoblats}: si gRPC es bloqueja, el WebSocket no s'atura.
        \item Frontend \textbf{mai fa polling}: tot és \textit{push}.
      \end{itemize}
    \end{column}

    \begin{column}{0.50\textwidth}
      \begin{center}
        \vspace{-0.4cm}
        \includegraphics[width=\textwidth, height=0.84\textheight, keepaspectratio]{../figures/diagrames/alarms_data_flow.png}

% ---  DIAPOSITIVA — Alarmes 3/3  ---
\begin{frame}{Alarmes \small\textit{3/3}}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \small
      \textbf{Persistència — \texttt{la-db-engine}:}
      \begin{itemize}
        \item \texttt{la-core-alarm} $\xrightarrow{\text{gRPC}}$ \textbf{SQLite3} + \texttt{remote\_vfs}
        \item \texttt{remote\_vfs}: driver \textbf{VFS} personalitzat — tot I/O $\xrightarrow{\text{UDS}}$ \texttt{la-video-engine}
        \item \texttt{la-video-engine}: únic procés amb accés \textbf{raw} al disc físic
        \item \textbf{WAL} (\texttt{PRAGMA synchronous=NORMAL}): registre transaccional previ a cada escriptura — \textbf{ACID} davant talls de corrent
        \item SQL \textbf{Trigger}: poda automàtica $\leq$ \textbf{10.000} registres
      \end{itemize}
    \end{column}

    \begin{column}{0.41\textwidth}
      \small
      \textbf{Protocol \texttt{/ws/la-core-alarm/}:}
      \begin{itemize}
        \item[$\leftarrow$] \texttt{DB\_ENGINE\_ALARM\_NOTIFY}
        \item[$\rightarrow$] \texttt{ALARMS\_GET}
        \item[$\rightarrow$] \texttt{ALARM\_READ}\\
          {\footnotesize\textcolor{gray}{\{db\_id: "1054", read: true\}}}
        \item[$\leftarrow$] \texttt{ALARM\_READ\_SUCCESS}\\
          {\footnotesize\textcolor{gray}{broadcast $\to$ tots els operadors}}
      \end{itemize}

      \vspace{0.4em}
      \begin{exampleblock}{\small Garantia}
        \small Cap alarma perduda.\\
        Consistència multi-operador.
      \end{exampleblock}
    \end{column}
  \end{columns}
\end{frame}
