      \begin{beamercolorbox}[wd=\paperwidth,ht=0.5pt,dp=0pt]{footline separator}%
      \end{beamercolorbox}%
    }%
    \hbox{%
      \begin{beamercolorbox}[wd=.333333\paperwidth,ht=2.5ex,dp=1.2ex,left,leftskip=2ex]{}%

      \end{beamercolorbox}%
      \begin{beamercolorbox}[wd=.333333\paperwidth,ht=2.5ex,dp=1.2ex,right,rightskip=2ex]{}%
        \textcolor{gray!80}{\footnotesize \insertframenumber{} / \inserttotalframenumber}%
      \end{beamercolorbox}%
    }%
  }%
}
